如何使用ModelForm快速生成表单

2015-09-01
Django forms
评论

Django中可以自定义表单，并且几乎和Models.py定义的方法一样。如果在Models.py中有些字段刚好表单中也需要，怎么办呢？

Django1.7版本中django.forms中有一个ModelForm类，可以快速把所需要的字段转换为表单字段。

#models.py
from django.forms import ModelForm
from django.db import models
from django.contrib.auth.models import AbstractUser

class User(AbstractUser):
    phone_num=models.CharFeild(max_length=100,required='required' verbose_name="手机号")

class UserForm (ModelForm):
    class Meta:
        model=User 
        fields=['username','phone_num']  #将所需要的字段放到fields的list中就可以了

#views.py
from myapp.models import UserForm
def register(request):
    if request.method=='POST':
        form=UserForm(request.POST)
        if form.is_valid():
            form.save() #保存表单传递过来的数据
    form=UserForm()  #生成表单实例
    return render_to_response('register.html',{'form':form})
